├── runtime-dom // 运行时dom相关api
├── runtime-test // 运行时测试
├── server-renderer // 服务端渲染
├── sfc-playground // 单文件在线调试器
├── shared // 内部共享工具库
├── size-check // 简单应用，用来测试代码体积
├── template-explorer
├── vue // vue入口文件
└── vue-compat
